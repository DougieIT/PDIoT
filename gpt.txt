private fun normalizeArray(inputArray: Array<Array<FloatArray>>): Array<Array<FloatArray>> {
    // Ensure the input array has the correct shape (1, 50, 3)
    require(inputArray.size == 1 && inputArray[0].size == 50 && inputArray[0][0].size == 3) {
        "Input array must have shape (1, 50, 3)"
    }

    // Normalize each (x, y, z) triplet in the array
    return Array(1) { i -> // First dimension is 1
        Array(50) { j -> // Second dimension is 50
            val (x, y, z) = inputArray[i][j]
            normalize(x, y, z) // Apply the normalization function
        }
    }
}
